 useEffect(() => {
    const init_input = {...INIT_FORMINPUT};
    const query = new URLSearchParams(props.location.search);
    for(let param of query.entries()) {
      if(param[0] !== 'page') {
        if(param[0] === 'species') {
          init_input['species'] = parseInt(param[1]);
        } else if(param[0] == 'name') {
          init_input['name'] = param[1];
        }else {
          const arrayString = param[1].slice(1, param[1].length - 1);
          let arr = arrayString.split(',');
          if(arr.length > 0) {
            arr = arr.map(a => parseInt(a));
            init_input[param[0]] = arr;
          }
        }
      }
    }
    setFormInput({...init_input});
  }, []);
